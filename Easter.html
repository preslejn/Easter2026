  const k = "easter_step";

  let step = Number(localStorage.getItem(k)) || 0;


  const clue = document.getElementById("clue");

  const msg = document.getElementById("msg");

  const prog = document.getElementById("progress");

  const input = document.getElementById("code");


  function render() {

    prog.textContent = `Step ${step+1} of ${steps.length}`;

    clue.textContent = steps[step].clue;

    msg.textContent = "";

    input.value = "";

  }


  async function sha256(t) {

    const b = new TextEncoder().encode(t);

    const h = await crypto.subtle.digest("SHA-256", b);

    return [...new Uint8Array(h)].map(x=>x.toString(16).padStart(2,"0")).join("");

  }


  document.getElementById("btn").onclick = async () => {

    const val = input.value.trim().toUpperCase();

    if (!val) return;

    const h = await sha256(val);

    if (h === steps[step].hash) {

      if (step === steps.length-1) {

        clue.textContent = `🏆 CONGRATULATIONS!\n\nYou finished the hunt!\n\nYour prize is located:\n${PRIZE_LOCATION}`;

        prog.textContent = "Complete!";

        msg.textContent = "";

        return;

      }

      step++;

      localStorage.setItem(k, step);

      render();

    } else {

      msg.textContent = "❌ Incorrect code — search the room again!";

    }

  };


  document.getElementById("resetBtn").onclick = () => {

    localStorage.removeItem(k);

    step = 0;

    render();

  };


  render();
